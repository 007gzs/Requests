 81:

 84:             $url = self::format_get($url, $data);

109:

148:     /**

156:         // Why do we do this? cURL will send both the final response and any
157:         // interim responses, such as a 100 Continue. We don't need that.
158:         // (We may want to keep this somewhere just in case)
159:         if ($this->done_headers) {

171:     /**
172:      * Format a URL given GET data
173:      *
174:      * @param string $url
175:      * @param array|object $data Data to build query using, see {@see http://php.net/http_build_query}
176:      * @return string URL with data
177:      */
178:     protected static function format_get($url, $data) {
179:         if (!empty($data)) {
180:             $url_parts = parse_url($url);
181:             if (empty($url_parts['query'])) {
182:                 $query = $url_parts['query'] = '';
183:             }
184:             else {
185:                 $query = $url_parts['query'];
186:             }
187: 
188:             $query .= '&' . http_build_query($data, null, '&');
189:             $query = trim($query, '&');
190: 
191:             if (empty($url_parts['query'])) {
192:                 $url .= '?' . $query;
193:             }
194:             else {
195:                 $url = str_replace($url_parts['query'], $query, $url);
196:             }
197:         }
198:         return $url;
199:     }
